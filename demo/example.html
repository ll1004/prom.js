Prom.all usage
		function task1(){
			return new Prom(function(resolve,reject){
				setTimeout(function(){
					resolve('A');
				},2000)
			})
		}
		function task2(){
			return new Prom(function(resolve,reject){
				setTimeout(function(){
					resolve('B');
				},3000)
			})
		}
		function task3(){
			return new Prom(function(resolve,reject){
				setTimeout(function(){
					resolve('C');
				},4000)
			})
		}
		Prom.all([task1(),task2(),task3()]).then(function(dataList){
			alert(dataList)
		}).catch(function(data){
			alert(data)
		})
